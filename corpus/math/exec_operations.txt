# EXEC OPERATIONS - Direct CPU Computation
# These patterns trigger EXEC nodes for actual computation

# Addition
ADD → NUMBER → NUMBER → EXEC_ADD → CPU → RESULT
SUM → EXEC_ADD → COMPUTE → VALUE
PLUS → EXEC_ADD → DIRECT → ANSWER

# Subtraction
SUBTRACT → NUMBER → NUMBER → EXEC_SUB → CPU → RESULT
MINUS → EXEC_SUB → COMPUTE → DIFFERENCE
TAKE_AWAY → EXEC_SUB → DIRECT → ANSWER

# Multiplication
MULTIPLY → NUMBER → NUMBER → EXEC_MUL → CPU → RESULT
TIMES → EXEC_MUL → COMPUTE → PRODUCT
X → EXEC_MUL → DIRECT → ANSWER

# Division
DIVIDE → NUMBER → NUMBER → EXEC_DIV → CPU → RESULT
DIVIDED_BY → EXEC_DIV → COMPUTE → QUOTIENT
SLASH → EXEC_DIV → DIRECT → ANSWER

# Bitwise Operations
AND → NUMBER → NUMBER → EXEC_AND → CPU → RESULT
OR → NUMBER → NUMBER → EXEC_OR → CPU → RESULT
XOR → NUMBER → NUMBER → EXEC_XOR → CPU → RESULT
NOT → NUMBER → EXEC_NOT → CPU → RESULT

# Comparison
COMPARE → NUMBER → NUMBER → EXEC_CMP → CPU → RESULT
GREATER_THAN → EXEC_CMP → COMPUTE → BOOLEAN
LESS_THAN → EXEC_CMP → COMPUTE → BOOLEAN
EQUAL → EXEC_CMP → COMPUTE → BOOLEAN
