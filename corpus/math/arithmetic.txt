# ARITHMETIC FOUNDATION
# Basic mathematical operations and patterns

# Addition
ADD → NUMBER → NUMBER → RESULT
SUM → ADD → TOTAL
PLUS → ADD → EQUALS

# Subtraction  
SUBTRACT → NUMBER → NUMBER → RESULT
MINUS → SUBTRACT → DIFFERENCE
TAKE_AWAY → SUBTRACT → RESULT

# Multiplication
MULTIPLY → NUMBER → NUMBER → PRODUCT
TIMES → MULTIPLY → RESULT
X → MULTIPLY → RESULT

# Division
DIVIDE → NUMBER → NUMBER → QUOTIENT
DIVIDED_BY → DIVIDE → RESULT
SLASH → DIVIDE → RESULT

# Number relationships
NUMBER → GREATER_THAN → NUMBER
NUMBER → LESS_THAN → NUMBER
NUMBER → EQUALS → NUMBER
NUMBER → COMPARE → RESULT

# Order of operations
EXPRESSION → PARENTHESES → EVALUATE → RESULT
EXPRESSION → MULTIPLY_DIVIDE → ADD_SUBTRACT → RESULT
OPERATION → ORDER → RESULT
